145     /**
146      * Check the given tld. Assure that:
147      * <ul>
148      * <li>Any tag class is loadable</li>
149      * <li>the tag class has a setter for any of the declared attribute</li>
150      * <li>the type declared in the dtd for an attribute (if any) matches the type accepted by the getter</li>
151      * </ul>
152      * @see org.apache.maven.reporting.AbstractMavenReportRenderer#renderBody()
153      */
154     @Override
155     protected void renderBody()
156     {
157         sink.body();
158         startSection(getMessageString("Validate.h1")); //$NON-NLS-1$
159         paragraph(getMessageString("Validate.into1")); //$NON-NLS-1$
160         paragraph(getMessageString("Validate.intro2")); //$NON-NLS-1$
161 
162         sink.list();
163         for (int j = 0; j < tlds.length; j++)
164         {
165 
166             sink.listItem();
167             sink.link("#" + tlds[j].getFilename()); //$NON-NLS-1$
168             sink.text(MessageFormat.format(getMessageString("Validate.listitem.tld"), new Object[]{ //$NON-NLS-1$
169                 StringUtils.defaultIfEmpty(tlds[j].getName(), tlds[j].getShortname()), tlds[j].getFilename() }));
170             sink.link_();
171             sink.text(getMessageString("Validate.listitem.uri") + tlds[j].getUri()); //$NON-NLS-1$
172 
173             sink.listItem_();
174         }
175         sink.list_();
176 
177         endSection();
178 
179         for (int j = 0; j < tlds.length; j++)
180         {
181             checkTld(tlds[j]);
182         }
183 
184         sink.body_();
185     }

187     /**
188      * Checks a single tld and returns validation results.
189      * @param tld Tld
190      */
191     private void checkTld(Tld tld)
192     {
193         // new section for each tld
194         sink.anchor(tld.getFilename());
195         sink.anchor_();
196         startSection(tld.getName() + " " + tld.getFilename()); //$NON-NLS-1$
197 
198         doTags(tld.getTags(), tld.getShortname());
199         doFunctions(tld.getFunctions(), tld.getShortname());
200 
201         endSection();
202     }

204     /**
205      * @param tld
206      */
207     private void doTags(Tag[] tags, String shortname)
208     {
209         if (tags != null && tags.length > 0)
210         {
211             for (int j = 0; j < tags.length; j++)
212             {
213                 Tag tldItem = tags[j];
214 
215                 checkTag(shortname, tldItem);
216             }
217         }
218     }

220     private void doFunctions(ELFunction[] tags, String shortname)
221     {
222         if (tags != null && tags.length > 0)
223         {
224 
225             startSection("EL functions");
226 
227             startTable();
228 
229             tableHeader(new String[]{
230                 getMessageString("Validate.header.validated"), "function", getMessageString("Validate.header.class"), getMessageString("Validate.header.signature") }); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$ //$NON-NLS-4$
231 
232             for (int j = 0; j < tags.length; j++)
233             {
234                 ELFunction tldItem = tags[j];
235                 checkFunction(shortname, tldItem);
236 
237             }
238 
239             endTable();
240 
241             endSection();
242         }
243     }

245     /**
246      * @param shortname
247      * @param tldItem
248      */
249     private void checkFunction(String prefix, ELFunction tag)
250     {
251 
252         String className = tag.getFunctionClass();
253 
254         boolean found = true;
255 
256         ClassLoader currentClassLoader = Thread.currentThread().getContextClassLoader();
257         Thread.currentThread().setContextClassLoader(this.projectClassLoader);
258 
259         try
260         {
261             Class<?> functionClass = Class.forName(className, true, this.projectClassLoader);
262 
263             String fullSignature = tag.getFunctionSignature();
264             String paramsString = tag.getParameters();
265             String returnvalue = null;
266 
267             String methodName = StringUtils.trim(StringUtils.substringBefore(fullSignature, "("));
268             if (StringUtils.contains(methodName, " "))
269             {
270                 returnvalue = StringUtils.substringBefore(methodName, " ");
271                 methodName = StringUtils.substringAfter(methodName, " ");
272             }
273 
274             String[] params = StringUtils.split(paramsString, ",");
275 
276             List<Class<?>> parClasses = new ArrayList<>(params.length);
277 
278             for (String stringClass : params)
279             {
280                 parClasses.add(Class.forName(StringUtils.trim(stringClass), true, this.projectClassLoader));
281             }
282 
283             Method method = functionClass.getMethod(methodName, parClasses.toArray(new Class<?>[0]));
284 
285             Class< ? > returnType = method.getReturnType();
286 
287             if (!(returnvalue == null || returnType.getCanonicalName().equals(returnvalue)))
288             {
289                 found = false;
290             }
291         }
292         catch (Throwable e)
293         {
294             found = false;
295         }
296 
297         Thread.currentThread().setContextClassLoader(currentClassLoader);
298 
299         sink.tableRow();
300 
301         sink.tableCell();
302         figure(found ? ICO_SUCCESS : ICO_ERROR);
303         sink.tableCell_();
304 
305         tableCell(prefix + ":" + tag.getName() + "()");
306         tableCell(className);
307         tableCell(tag.getFunctionSignature());
308 
309         sink.tableRow_();
310 
311     }

313     /**
314      * Checks a single tag and returns validation results.
315      * @param tag Tag
316      */
317     private void checkTag(String prefix, Tag tag)
318     {
319 
320         // new subsection for each tag
321         startSection("<" + prefix + ":" + tag.getName() + ">"); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
322 
323         String className = tag.getTagClass();
324 
325         startTable();
326 
327         tableHeader(new String[]{
328             getMessageString("Validate.header.found"), getMessageString("Validate.header.loadable"), getMessageString("Validate.header.extends"), getMessageString("Validate.header.class") }); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$ //$NON-NLS-4$
329 
330         boolean found = true;
331         boolean loadable = true;
332         boolean extend = true;
333 
334         Object tagObject = null;
335         ClassLoader currentClassLoader = Thread.currentThread().getContextClassLoader();
336         Thread.currentThread().setContextClassLoader(this.projectClassLoader);
337 
338         try
339         {
340             Class<?> tagClass = Class.forName(className, true, this.projectClassLoader);
341 
342             // extend only true, if tagClass derives from TagSupport or derives from SimpleTag
343             extend = tagSupportClass.isAssignableFrom(tagClass)
344                 || simpleTagClass != null && simpleTagClass.isAssignableFrom(tagClass);
345 
346             try
347             {
348                 tagObject = tagClass.getDeclaredConstructor().newInstance();
349             }
350             catch (Throwable e)
351             {
352                 loadable = false;
353             }
354 
355         }
356         catch (ClassNotFoundException e)
357         {
358             found = false;
359             loadable = false;
360             extend = false;
361         }
362         catch (NoClassDefFoundError e)
363         {
364             found = false;
365             loadable = false;
366             extend = false;
367         }
368 
369         Thread.currentThread().setContextClassLoader(currentClassLoader);
370 
371         TagAttribute[] attributes = tag.getAttributes();
372 
373         sink.tableRow();
374 
375         sink.tableCell();
376         figure(found ? ICO_SUCCESS : ICO_ERROR);
377         sink.tableCell_();
378 
379         sink.tableCell();
380         figure(loadable ? ICO_SUCCESS : ICO_ERROR);
381         sink.tableCell_();
382 
383         sink.tableCell();
384         figure(extend ? ICO_SUCCESS : ICO_ERROR);
385         sink.tableCell_();
386 
387         tableCell(className);
388 
389         sink.tableRow_();
390 
391         if (tag.getTeiClass() != null)
392         {
393             checkTeiClass(tag.getTeiClass());
394         }
395 
396         endTable();
397 
398         if (tagObject != null && attributes.length > 0)
399         {
400 
401             startTable();
402             tableHeader(new String[]{
403                 StringUtils.EMPTY,
404                 getMessageString("Validate.header.attributename"), getMessageString("Validate.header.tlddeclares"), getMessageString("Validate.header.tagdeclares") }); //$NON-NLS-1$//$NON-NLS-2$ //$NON-NLS-3$ //$NON-NLS-4$
405 
406             for (int j = 0; j < attributes.length; j++)
407             {
408                 checkAttribute(tagObject, attributes[j]);
409             }
410 
411             endTable();
412         }
413         endSection();
414     }

416     /**
417      * Check a declared TagExtraInfo class.
418      * @param className TEI class name
419      */
420     private void checkTeiClass(String className)
421     {
422 
423         boolean found = true;
424         boolean loadable = true;
425         boolean extend = true;
426 
427         Class<?> teiClass = null;
428         try
429         {
430             teiClass = Class.forName(className, true, this.projectClassLoader);
431 
432             if (tagExtraInfoClass == null || !tagExtraInfoClass.isAssignableFrom(teiClass))
433             {
434                 extend = false;
435             }
436 
437             try
438             {
439                 teiClass.getDeclaredConstructor().newInstance();
440             }
441             catch (Throwable e)
442             {
443                 loadable = false;
444             }
445         }
446         catch (ClassNotFoundException e)
447         {
448             found = false;
449             loadable = false;
450             extend = false;
451         }
452 
453         catch (NoClassDefFoundError e)
454         {
455             found = false;
456             loadable = false;
457             extend = false;
458         }
459 
460         sink.tableRow();
461 
462         sink.tableCell();
463         figure(found ? ICO_SUCCESS : ICO_ERROR);
464         sink.tableCell_();
465 
466         sink.tableCell();
467         figure(loadable ? ICO_SUCCESS : ICO_ERROR);
468         sink.tableCell_();
469 
470         sink.tableCell();
471         figure(extend ? ICO_SUCCESS : ICO_ERROR);
472         sink.tableCell_();
473 
474         sink.tableCell();
475         sink.text(className);
476         sink.tableCell_();
477 
478         sink.tableRow_();
479     }

481     /**
482      * Checks a single attribute and returns validation results.
483      * @param tag tag handler instance
484      * @param attribute TagAttribute
485      */
486     private void checkAttribute(Object tag, TagAttribute attribute)
487     {
488 
489         String tldType = attribute.getType();
490         String tldName = attribute.getName();
491         Class<?> tagType = null;
492         String tagTypeName = null;
493 
494         List<ValidationError> validationErrors = new ArrayList<>(3);
495 
496         if (!PropertyUtils.isWriteable(tag, tldName))
497         {
498             validationErrors.add(new ValidationError(ValidationError.LEVEL_ERROR,
499                 getMessageString("Validate.error.setternotfound"))); //$NON-NLS-1$
500         }
501 
502         // don't check if setter is missing
503         if (validationErrors.isEmpty())
504         {
505 
506             try
507             {
508                 tagType = PropertyUtils.getPropertyType(tag, tldName);
509             }
510             catch (Throwable e)
511             {
512                 // should never happen, since we already checked the writable property
513                 log.warn(e);
514             }
515             tagTypeName = tagType == null ? StringUtils.EMPTY : tagType.getName();
516 
517             if (tldType != null && tagType != null)
518             {
519                 Class<?> tldTypeClass = getClassFromName(tldType);
520 
521                 if (!tagType.isAssignableFrom(tldTypeClass))
522                 {
523 
524                     validationErrors.add(new ValidationError(ValidationError.LEVEL_ERROR, MessageFormat.format(
525                         getMessageString("Validate.error.attributetypemismatch"), new Object[]{ //$NON-NLS-1$
526                         tldType, tagType.getName() })));
527                 }
528             }
529         }
530 
531         // don't check if we already know type is different
532         if (validationErrors.isEmpty())
533         {
534 
535             if (tldType != null && !tldType.equals(tagType.getName()))
536             {
537                 validationErrors.add(new ValidationError(ValidationError.LEVEL_WARNING, MessageFormat.format(
538                     getMessageString("Validate.error.attributetypeinexactmatch"), new Object[]{ //$NON-NLS-1$
539                     tldType, tagType.getName() })));
540             }
541             else if (tldType == null && !String.class.equals(tagType))
542             {
543                 validationErrors.add(new ValidationError(ValidationError.LEVEL_INFO,
544                     getMessageString("Validate.error.attributetype"))); //$NON-NLS-1$
545             }
546         }
547 
548         sink.tableRow();
549 
550         sink.tableCell();
551 
552         int figure = ICO_SUCCESS;
553 
554         for (Iterator<ValidationError> iter = validationErrors.iterator(); iter.hasNext();)
555         {
556             ValidationError error = iter.next();
557             if (error.getLevel() == ValidationError.LEVEL_ERROR)
558             {
559                 figure = ICO_ERROR;
560             }
561             else if (figure == ICO_SUCCESS) // warning
562             {
563                 figure = ICO_WARNING;
564             }
565 
566         }
567 
568         figure(figure);
569         sink.tableCell_();
570 
571         sink.tableCell();
572         sink.text(tldName);
573 
574         Iterator<ValidationError> iterator = validationErrors.iterator();
575         while (iterator.hasNext())
576         {
577             ValidationError error = iterator.next();
578             sink.lineBreak();
579             if (error.getLevel() == ValidationError.LEVEL_ERROR)
580             {
581                 sink.bold();
582             }
583             sink.text(error.getText());
584             if (error.getLevel() == ValidationError.LEVEL_ERROR)
585             {
586                 sink.bold_();
587 
588             }
589         }
590 
591         sink.tableCell_();
592 
593         sink.tableCell();
594         if (tldType != null)
595         {
596             sink.text(StringUtils.substringAfter(tldType, "java.lang.")); //$NON-NLS-1$
597         }
598         sink.tableCell_();
599 
600         tableCell(StringUtils.substringAfter(tagTypeName, "java.lang.")); //$NON-NLS-1$
601 
602         sink.tableRow_();
603 
604     }

606     private void figure(int type)
607     {
608         String text;
609         String src;
610 
611         switch (type)
612         {
613             case ICO_ERROR :
614                 text = getMessageString("Validate.level.error"); //$NON-NLS-1$
615                 src = IMAGE_ERROR_SRC;
616                 break;
617             case ICO_WARNING :
618                 text = getMessageString("Validate.level.warning"); //$NON-NLS-1$
619                 src = IMAGE_WARNING_SRC;
620                 break;
621             case ICO_INFO :
622                 text = getMessageString("Validate.level.info"); //$NON-NLS-1$
623                 src = IMAGE_INFO_SRC;
624                 break;
625             default :
626                 text = getMessageString("Validate.level.success"); //$NON-NLS-1$
627                 src = IMAGE_SUCCESS_SRC;
628                 break;
629         }
630 
631         sink.figure();
632         sink.figureGraphics(src);
633         sink.figureCaption();
634         sink.text(text);
635         sink.figureCaption_();
636         sink.figure_();
637     }

639     /**
640      * returns a class from its name, handling primitives.
641      * @param className clss name
642      * @return Class istantiated using Class.forName or the matching primitive.
643      */
644     private Class<?> getClassFromName(String className)
645     {
646 
647         Class<?> tldTypeClass = tryGettingPrimitiveClass(className);
648 
649         if (tldTypeClass == null)
650         {
651             // not a primitive type
652             try
653             {
654                 if (isArrayClassName(className))
655                 {
656                     tldTypeClass = getArrayClass(className);
657                 }
658                 else
659                 {
660                     tldTypeClass = Class.forName(className, true, this.projectClassLoader);
661                 }
662             }
663             catch (ClassNotFoundException e)
664             {
665                 log.error(MessageFormat.format(Messages.getString("Validate.error.unabletofindclass"), //$NON-NLS-1$
666                     new Object[]{className }));
667             }
668         }
669         return tldTypeClass;
670     }
671 
672     private Class<?> tryGettingPrimitiveClass(String className)
673     {
674         if ("int".equals(className)) //$NON-NLS-1$
675         {
676             return int.class;
677         }
678         if ("long".equals(className)) //$NON-NLS-1$
679         {
680             return long.class;
681         }
682         if ("double".equals(className)) //$NON-NLS-1$
683         {
684             return double.class;
685         }
686         if ("boolean".equals(className)) //$NON-NLS-1$
687         {
688             return boolean.class;
689         }
690         if ("char".equals(className)) //$NON-NLS-1$
691         {
692             return char.class;
693         }
694         if ("byte".equals(className)) //$NON-NLS-1$
695         {
696             return byte.class;
697         }
698 
699         return null;
700     }
701 
702     private boolean isArrayClassName(String className)
703     {
704         return className.endsWith("[]");
705     }
706 
707     private Class<?> getArrayClass(String className) throws ClassNotFoundException
708     {
709         String elementClassName = StringUtils.replace(className, "[]", "");
710         Class<?> elementClass = tryGettingPrimitiveClass(elementClassName);
711         if (elementClass == null)
712         {
713             elementClass = Class.forName(elementClassName);
714         }
715         return Array.newInstance(elementClass, 0).getClass();
716     }

718     static class ValidationError
719     {
720 
721         public static final int LEVEL_INFO = 1;
722 
723         public static final int LEVEL_WARNING = 2;
724 
725         public static final int LEVEL_ERROR = 3;
726 
727         private int level;
728 
729         private String text;
730 
731         /**
732          * @param level
733          * @param text
734          */
735         public ValidationError(int level, String text)
736         {
737             this.level = level;
738             this.text = text;
739         }
740 
741         /**
742          * Getter for <code>level</code>.
743          * @return Returns the level.
744          */
745         public int getLevel()
746         {
747             return this.level;
748         }
749 
750         /**
751          * Setter for <code>level</code>.
752          * @param level The level to set.
753          */
754         public void setLevel(int level)
755         {
756             this.level = level;
757         }
758 
759         /**
760          * Getter for <code>text</code>.
761          * @return Returns the text.
762          */
763         public String getText()
764         {
765             return this.text;
766         }
767 
768         /**
769          * Setter for <code>text</code>.
770          * @param text The text to set.
771          */
772         public void setText(String text)
773         {
774             this.text = text;
775         }
776     }

66  /**
67   * Generates tld files from directories of jsp 2.0 tag files.
68   * @author Fabrizio Giustina
69   * @version $Id: TldGenerateMojo.java 217 2014-08-15 20:50:32Z fgiust $

71  @Mojo(name="tldgenerate", defaultPhase=LifecyclePhase.GENERATE_RESOURCES)
72  public class TldGenerateMojo extends AbstractMojo

75      /**
76       * Directory containing tag files. Subdirectories are also processed.
77       */
78      @Parameter(defaultValue="src/main/resources/META-INF/tags/")
79      private File tagDir;
80  
81      /**
82       * Output dir for tld files.
83       */
84      @Parameter(defaultValue="${project.build.outputDirectory}/META-INF", required=true)
85      private File outputDir;

87      /**
88       * Version added to tld files, defaults to project version.
89       */
90      @Parameter(property="project.version", required=true)
91      private String version;

93      /**
94       * Detailed configuration for taglibs for tld generation. Starting with version 2.4 you can configure multiple
95       * taglibs with this attribute, and for each taglib you can add both tagfiles dir than classes with EL functions
96       * (note that EL function support is preliminary, the resulting tld does not include anything in the "description"
97       * attribute (that would require parsing javadocs from sources.
98       * 
99       * <pre>
100      *    &lt;taglibs>
101      *      &lt;taglib>
102      *        &lt;description>A test tld that contains functions&lt;/description>
103      *        &lt;shortName>test&lt;/shortName>
104      *        &lt;uri>testuri&lt;/uri>
105      *        &lt;outputname>testtaglib&lt;/outputname>
106      *        &lt;functionClasses>
107      *          &lt;functionClass>org.apache.commons.lang.StringUtils&lt;/functionClass>
108      *        &lt;/functionClasses>
109      *        &lt;tagdir>src/tagfiles&lt;/tagdir>
110      *      &lt;/taglib>
111      *    &lt;/taglibs>
112      * </pre>
113      */
114     @Parameter
115     private List<Taglib> taglibs;

117     /**
118      * @see org.apache.maven.plugin.Mojo#execute()
119      */
120     public void execute() throws MojoExecutionException, MojoFailureException
121     {
122         getLog().debug(MessageFormat.format(Messages.getString("Taglib.generating.tld"), //$NON-NLS-1$
123             new Object[]{tagDir.getAbsolutePath() }));
124 
125         try
126         {
127             generateTlds();
128         }
129         catch (IOException e)
130         {
131             throw new MojoExecutionException(e.getMessage(), e);
132         }
133 
134     }

136     /**
137      * @throws IOException
138      * @throws MojoExecutionException
139      */
140     private void generateTlds() throws IOException, MojoExecutionException
141     {
142 
143         List<Taglib> taglibsList = new ArrayList<Taglib>();
144 
145         if (taglibs != null)
146         {
147             for (Taglib taglib : taglibs)
148             {
149                 // a drink to get the old-style embedding tests work,
150                 // I should migrate to the new IT test infrastructure
151                 Taglib tlib = new Taglib();
152                 try
153                 {
154                     PropertyUtils.copyProperties(tlib, taglib);
155                 }
156                 catch (Throwable e)
157                 {
158                     throw new MojoExecutionException(e.getMessage(), e);
159                 }
160                 taglibsList.add(tlib);
161             }
162         }
163 
164         if (taglibsList == null || taglibsList.isEmpty())
165         {
166             taglibsList = new ArrayList<Taglib>();
167 
168             // old (pre 2.4) behavior, create taglib configurations from dir
169             if (tagDir.isDirectory())
170             {
171                 // handle tag files. Add any directory containing .tag or .tagx files
172                 List<File> tags = FileUtils.getFiles(tagDir, "**/*.tag", null); //$NON-NLS-1$
173                 tags.addAll(FileUtils.getFiles(tagDir, "**/*.tagx", null)); //$NON-NLS-1$
174 
175                 if (!tags.isEmpty())
176                 {
177                     Set<File> directories = new HashSet<File>();
178                     for (File tag : tags)
179                     {
180                         directories.add(tag.getParentFile());
181                     }
182 
183                     for (File dir : directories)
184                     {
185                         Taglib tlib = new Taglib();
186                         tlib.setTagdir(dir);
187                         tlib.setShortName(dir.getName());
188                         tlib.setOutputname(dir.getName() + ".tld");
189                         tlib.setUri(dir.getName());
190                         tlib.setShortName("Tag library for tag file directory " + dir.getName());
191                         taglibsList.add(tlib);
192                     }
193                 }
194                 else
195                 {
196                     getLog().warn(MessageFormat.format(Messages.getString("Taglib.generating.notfound"), //$NON-NLS-1$
197                         new Object[]{tagDir.getAbsolutePath() }));
198                 }
199             }
200         }
201 
202         try
203         {
204             for (Taglib taglib : taglibsList)
205             {
206                 doTaglib(taglib);
207             }
208         }
209         catch (Exception e)
210         {
211             throw new MojoExecutionException(e.getMessage(), e);
212         }
213     }

215     /**
216      * @param taglib
217      * @throws MojoExecutionException
218      * @throws FileNotFoundException
219      * @throws IOException
220      */
221     private void doTaglib(Taglib taglib) throws MojoExecutionException, FileNotFoundException, IOException
222     {
223         Document doc = getTLDDocument(taglib, XmlHelper.getDocumentBuilder());
224 
225         if (taglib.getShortName() == null)
226         {
227             throw new MojoExecutionException("Missing \"shortName\" parameter for taglib " + taglib);
228         }
229 
230         String tldName = taglib.getOutputname();
231         if (tldName == null)
232         {
233             tldName = taglib.getShortName() + ".tld";
234         }
235         if (!tldName.endsWith(".tld"))
236         {
237             tldName = tldName + ".tld";
238         }
239 
240         File outputFile = new File(outputDir, tldName);
241         if (!(outputDir.mkdirs() || outputDir.isDirectory())) {
242             throw new IOException("Unable to create output directory " + outputDir.getAbsolutePath());
243         }
244 
245         getLog().info(MessageFormat.format(Messages.getString("Taglib.generating.file"), //$NON-NLS-1$
246             new Object[]{tldName, taglib.getShortName() }));
247 
248         try (FileOutputStream fos = new FileOutputStream(outputFile)) 
249         {
250             XMLOutputter outputter = new XMLOutputter(Format.getPrettyFormat());
251             outputter.output(new DOMBuilder().build(doc), fos);
252         }
253     }

255     protected Document getTLDDocument(Taglib taglib, DocumentBuilder documentBuilder)
256 
257     {
258         Document result = documentBuilder.newDocument();
259 
260         Element taglibElement = createRootTaglibNode(result, taglib.getDescription(), taglib.getShortName(), taglib
261             .getUri());
262 
263         if (taglib.getTagdir() != null && taglib.getTagdir().isDirectory())
264         {
265 
266             String path = taglib.getTagdir().getAbsolutePath().replace(File.separatorChar, '/');
267             int index = path.indexOf("/META-INF/");
268             if (index != -1)
269             {
270                 path = path.substring(index);
271             }
272             else
273             {
274                 path = "unknown";
275             }
276             if (!path.endsWith("/"))
277             {
278                 path += "/";
279             }
280 
281             File[] files = taglib.getTagdir().listFiles();
282             for (int j = 0; files != null && j < files.length; j++)
283             {
284                 File tag = files[j];
285 
286                 if (!tag.isDirectory()
287                     && (tag.getName().toLowerCase().endsWith(".tag") || tag.getName().toLowerCase().endsWith(".tagx")))
288                 {
289                     String tagName = tag.getName().substring(0, tag.getName().lastIndexOf('.'));
290                     String tagPath = path + tag.getName();
291 
292                     Element tagFileElement = result.createElement("tag-file");
293                     Element nameElement = result.createElement("name");
294                     nameElement.appendChild(result.createTextNode(tagName));
295                     tagFileElement.appendChild(nameElement);
296                     Element pathElement = result.createElement("path");
297                     pathElement.appendChild(result.createTextNode(tagPath));
298                     tagFileElement.appendChild(pathElement);
299                     taglibElement.appendChild(tagFileElement);
300 
301                     // tag-file Subelements
302                     //
303                     // description (optional) A description of the tag.
304                     // display-name (optional) Name intended to be displayed by tools.
305                     // icon (optional) Icon that can be used by tools.
306                     // name The unique tag name.
307                     // path Where to find the tag file implementing this tag, relative to the root of the web
308                     // application or
309                     // the root
310                     // of the JAR file for a tag library packaged in a JAR. This must begin with /WEB-INF/tags/ if the
311                     // tag
312                     // file resides
313                     // in the WAR, or /META-INF/tags/ if the tag file resides in a JAR.
314                     // example (optional) Informal description of an example use of the tag.
315                     // tag-extension (optional) Extensions that provide extra information about the tag for tools.
316 
317                     try (InputStream is = new FileInputStream(tag))
318                     {
319                         TagFile tagFile = TagFile.parse(is);
320 
321                         for (Directive directive : tagFile.getDirectives())
322                         {
323                             if ("tag".equals(directive.getDirectiveName()))
324                             {
325                                 for (Attribute attribute : directive.getAttributes())
326                                 {
327                                     if ("description".equals(attribute.getName())
328                                         || "display-name".equals(attribute.getName())
329                                         || "example".equals(attribute.getName()))
330                                     {
331                                         Element element = result.createElement(attribute.getName());
332                                         element.appendChild(result.createTextNode(attribute.getValue()));
333                                         tagFileElement.appendChild(element);
334                                     }
335 
336                                 }
337 
338                             }
339 
340                         }
341 
342                     }
343                     catch (IOException e)
344                     {
345                         // @todo Auto-generated catch block
346                         e.printStackTrace();
347                     }
348                     catch (ParseException e)
349                     {
350                         // @todo Auto-generated catch block
351                         e.printStackTrace();
352                     }
353 
354                 }
355 
356             }
357         }
358 
359         Map<String, Integer> duplicateFunctions = new HashMap<String, Integer>();
360 
361         if (taglib.getFunctionClasses() != null)
362         {
363             for (int j = 0; j < taglib.getFunctionClasses().length; j++)
364             {
365                 String functionClassString = taglib.getFunctionClasses()[j];
366 
367                 Class<?> functionClass = null;
368                 try
369                 {
370                     functionClass = Class.forName(functionClassString);
371 
372                 }
373                 catch (Throwable e)
374                 {
375                     getLog().error(
376                         "Unable to load function class "
377                             + functionClassString
378                             + ": "
379                             + e.getClass().getName()
380                             + " "
381                             + e.getMessage(),
382                         e);
383                     continue;
384                 }
385                 Method[] declaredMethods = functionClass.getDeclaredMethods();
386 
387                 for (int k = 0; k < declaredMethods.length; k++)
388                 {
389                     Method method = declaredMethods[k];
390                     if (!Modifier.isStatic(method.getModifiers()) || !Modifier.isPublic(method.getModifiers()))
391                     {
392                         // not a public static method
393                         continue;
394                     }
395 
396                     Element tagFileElement = result.createElement("function");
397 
398                     String functionName = method.getName();
399 
400                     if (duplicateFunctions.containsKey(functionName))
401                     {
402                         int currentNum = duplicateFunctions.get(functionName);
403                         duplicateFunctions.put(functionName, currentNum + 1);
404                         functionName = functionName + currentNum;
405                     }
406                     else
407                     {
408                         duplicateFunctions.put(functionName, 1);
409                     }
410 
411                     Element nameElement = result.createElement("name");
412                     nameElement.appendChild(result.createTextNode(functionName));
413                     tagFileElement.appendChild(nameElement);
414 
415                     Element classElement = result.createElement("function-class");
416                     classElement.appendChild(result.createTextNode(method.getDeclaringClass().getName()));
417                     tagFileElement.appendChild(classElement);
418 
419                     StringBuffer parameterTypesString = new StringBuffer();
420                     Class< ? >[] parameterTypes = method.getParameterTypes();
421 
422                     for (int p = 0; p < parameterTypes.length; p++)
423                     {
424                         Class< ? > param = parameterTypes[p];
425                         parameterTypesString.append(nonPrimitiveName(param.getCanonicalName()));
426                         if (parameterTypes.length - 1 > p)
427                         {
428                             parameterTypesString.append(", ");
429                         }
430                     }
431 
432                     Element signatureElement = result.createElement("function-signature");
433                     signatureElement.appendChild(result.createTextNode(nonPrimitiveName(method
434                         .getReturnType()
435                         .getCanonicalName())
436                         + " "
437                         + method.getName()
438                         + "("
439                         + parameterTypesString.toString()
440                         + ")"));
441                     tagFileElement.appendChild(signatureElement);
442 
443                     taglibElement.appendChild(tagFileElement);
444 
445                 }
446 
447             }
448         }
449 
450         return result;
451     }
452 
453     private String nonPrimitiveName(String string)
454     {
455         if ("int".equals(string))
456         {
457             return Integer.class.getName();
458         }
459         else if ("boolean".equals(string))
460         {
461             return Boolean.class.getName();
462         }
463         else if ("double".equals(string))
464         {
465             return Double.class.getName();
466         }
467         else if ("long".equals(string))
468         {
469             return Long.class.getName();
470         }
471         else if ("char".equals(string))
472         {
473             return Character.class.getName();
474         }
475 
476         return string;
477     }

479     protected Element createRootTaglibNode(Document result, String description, String shortName, String uri)
480     {
481         Element taglibElement = result.createElementNS("http://java.sun.com/xml/ns/javaee", "taglib");
482         taglibElement.setAttributeNS("http://www.w3.org/2000/xmlns/", "xmlns", "http://java.sun.com/xml/ns/javaee");
483         taglibElement.setAttributeNS(
484             "http://www.w3.org/2000/xmlns/",
485             "xmlns:xsi",
486             "http://www.w3.org/2001/XMLSchema-instance");
487         taglibElement.setAttributeNS(
488             "http://www.w3.org/2001/XMLSchema-instance",
489             "xsi:schemaLocation",
490             "http://java.sun.com/xml/ns/javaee http://java.sun.com/xml/ns/javaee/web-jsptaglibrary_2_1.xsd");
491         taglibElement.setAttribute("version", "2.1");
492         result.appendChild(taglibElement);
493 
494         Element descriptionElement = result.createElement("description");
495         descriptionElement.appendChild(result.createTextNode(description));
496         taglibElement.appendChild(descriptionElement);
497 
498         Element tlibVersionElement = result.createElement("tlib-version");
499         tlibVersionElement.appendChild(result.createTextNode(version));
500         taglibElement.appendChild(tlibVersionElement);
501 
502         Element shortNameElement = result.createElement("short-name");
503         shortNameElement.appendChild(result.createTextNode(shortName));
504         taglibElement.appendChild(shortNameElement);
505 
506         Element uriElement = result.createElement("uri");
507         uriElement.appendChild(result.createTextNode(uri));
508         taglibElement.appendChild(uriElement);
509 
510         return taglibElement;
511     }
